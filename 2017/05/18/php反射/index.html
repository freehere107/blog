<?php

namespace app;
//反射

trait future
{
    public function hello()
    {
        echo "hello";
    }
}

Class Daddy
{

}

/**
 * 构造函数
 * public ReflectionClass::__construct ( mixed $argument )
 */
interface B
{
}

interface C
{
}

Class A extends Daddy implements B, C
{
    private $sum = 20;
    private $name = 'free';
    const AGE = 55;
    protected $gender = 'female';
    public static $color = 'Red';

    function __construct()
    {
        $this->sum = 10;
        $this->name = 'A class';
        $this->gender = 'man';
    }

    public function run()
    {
        $this->sum++;
    }

    public function fly()
    {
        $this->sum = 999;
    }
}

$instant = new \ReflectionClass('app\\A');

// ReflectionClass::export — 导出一个类
$a = \ReflectionClass::export($instant);
print_r($a);

/** 输出
 * Class [ <user> class A ] {
 * @@ ReflectionClass.php A8-18
 *
 * - Constants [0] {
 * }
 *
 * - Static properties [0] {
 * }
 *
 * - Static methods [0] {
 * }
 *
 * - Properties [2] {
 * Property [ <default> private $sum ]
 * Property [ <default> private $name ]
 * }
 *
 * - Methods [1] {
 * Method [ <user, ctor> public method __construct ] {
 * @@ ReflectionClass.php 13 - 17
 * }
 * }
 * }*/

// ReflectionClass::getConstant — 获取定义过的一个常量
$getAGE = $instant->getConstant('AGE');
echo 'get param AGE ' . $getAGE . "\r\n"; //get param AGE 55

// ReflectionClass::getConstants — 获取一组常量
$getConstantArr = $instant->getConstants();
print_r($getConstantArr);
/**
 * Array
 * (
 * [AGE] => 55
 * )*/

// ReflectionClass::getConstructor 获取类的构造函数

// ReflectionClass::getDefaultProperties — 获取默认属性

$getDefaultPro = $instant->getDefaultProperties();
print_r($getDefaultPro);
/**
 * Array
 * (
 * [sum] => 20
 * [name] => free
 * [gender] => female
 * )
 */

// ReflectionClass::getDocComment — 获取文档注释

// ReflectionClass::getEndLine — 获取最后一行的行数

// ReflectionClass::getExtension — 根据已定义的类获取所在扩展的 ReflectionExtension 对象

// ReflectionClass::getExtensionName — 获取定义的类所在的扩展的名称，如果是用户定义的类，则返回 FALSE。
$getExtName = $instant->getExtensionName();
var_dump($getExtName); //bool(false)

// ReflectionClass::getFileName — 获取定义类的文件名,返回类所定义的文件名。如果这个类是在 PHP 核心或 PHP 扩展中定义的，则返回 FALSE。
$getFileName = $instant->getFileName();
var_dump($getFileName); //ReflectionClass.php

// ReflectionClass::getInterfaceNames — 获取接口（interface）名称
$getInterName = $instant->getInterfaceNames();
print_r($getInterName);
/**
 * Array
 * (
 * [0] => B
 * [1] => C
 * )
 */

// ReflectionClass::getInterfaces — 获取接口
$getInterface = $instant->getInterfaces();
print_r($getInterface);

/**
 * Array
 * (
 * [B] => ReflectionClass Object
 * (
 * [name] => B
 * )
 *
 * [C] => ReflectionClass Object
 * (
 * [name] => C
 * )
 * )
 */


// ReflectionClass::getMethod — 获取一个类方法的 ReflectionMethod。

$getRunMethod = $instant->getMethod('run');
print_r($getRunMethod);

/**
 * ReflectionMethod Object
 * (
 * [name] => run
 * [class] => A
 * )
 */

//ReflectionClass::getMethods — 获取方法的数组,包括构造函数

$getAllMethod = $instant->getMethods();
print_r($getAllMethod);

/**
 * Array
 * (
 * [0] => ReflectionMethod Object
 * (
 * [name] => __construct
 * [class] => A
 * )
 *
 * [1] => ReflectionMethod Object
 * (
 * [name] => run
 * [class] => A
 * )
 *
 * [2] => ReflectionMethod Object
 * (
 * [name] => fly
 * [class] => A
 * )
 * )
 */


// ReflectionClass::getModifiers — 获取类的修饰符

// ReflectionClass::getName — 获取类名
$getClassName = $instant->getName();
echo 'class name is ' . $getClassName . "\r\n"; // class name is app\A


// ReflectionClass::getNamespaceName — 获取命名空间的名称
$getNamespaceName = $instant->getNamespaceName();
echo 'Namespace name is ' . $getNamespaceName . "\r\n";


// ReflectionClass::getParentClass — 获取父类
$getClassDaddy = $instant->getParentClass();
print_r($getClassDaddy);
/**
 * ReflectionClass Object
 * (
 * [name] => app\Daddy
 * )
 */


// ReflectionClass::getProperties — 获取一组属性
/**
 * public array ReflectionClass::getProperties ([ int $filter ] )
 * filter 可选的过滤器，过滤为所需类型的属性。它使用 ReflectionProperty 常量 来配置，默认获取所有类型的属性。
